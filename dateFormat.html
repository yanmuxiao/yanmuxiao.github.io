一、日期格式化代码：

Date.prototype.format = function(fmt) { 
   var o = { 
      "M+" : this.getMonth()+1,                 //月份==>注意月的M是大写的 
      "d+" : this.getDate(),                    //日 
      "h+" : this.getHours(),                   //小时 
      "m+" : this.getMinutes(),                 //分 ==>注意分钟的m是小写的 
      "s+" : this.getSeconds(),                 //秒 
      "q+" : Math.floor((this.getMonth()+3)/3), //季度 
      "S"  : this.getMilliseconds()             //毫秒 
  }; 
  if(/(y+)/.test(fmt)) {
          fmt=fmt.replace(RegExp.$1, (this.getFullYear()+"").substr(4 - RegExp.$1.length)); 
  }
  for(var k in o) {
    if(new RegExp("("+ k +")").test(fmt)){
         fmt = fmt.replace(RegExp.$1, (RegExp.$1.length==1) ? (o[k]) : (("00"+ o[k]).substr((""+ o[k]).length)));
     }
  }
  return fmt; 
}

二、例子

1. console.log(new date()); ==> Tue May 23 2017 23:06:12 GMT+0800 (中国标准时间)

    console.log(new Date().format('yyyy-MM-dd'));  ==> 2017-06-23

    console.log(new Date().format('yy-MM-dd'));  ==> 2017-05-23

    console.log(new Date().format('yyyy-M-d'));  ==> 2017-5-23

    console.log('==>' + new Date().format('MM-dd')) ==> 05-23

    console.log('==>' + new Date().format('MM')) ==> 05

    console.log('==>' + new Date().format('dd')) ==> 23

    console.log(new date().format('yyyy/mm/dd'));  ==> 2017/06/23

2. console.log('==>' + new Date().format('yyyy-MM-dd hh:mm:ss'))
    ==> 2017-05-23 23:30:54

   console.log('==>' + new Date().format('yyyy-MM-dd h:m:s'))
    ==> 2017-05-23 23:33:3

   console.log('==>' + new Date().format('yyyy-MM-dd h:m'))
    ==> 2017-05-23 23:33
