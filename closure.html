一、自由变量 ```javascript var x = 10; // 全局作用域A中的变量x function fn1() { // 局部作用域fn1 var y = 20; console.log(x); function fn2() { // 局部作用域fn2 console.log(y); } } ``` 作用域链：全局作用域A ==> 局部作用域fn1 ==> 局部作用域fn2 我们称在局部作用域中通过作用域链使用的变量为自由变量，x是fn1中的自由变量，y是fn2中的自由变量。 二、局部作用域中自由变量的指向 ```javascript var x = 10; // 全局作用域A中的变量x function fn1() { // 局部作用域fn1 console.log(x); } var fn3 = function() { // 局部作用域fn3 var x = 30; fn1(); } fn3(); // 10 ``` 在局部作用域形成(或者说定义)时，局部作用域中自由变量的指向已经确定，与在哪里调用执行无关。 三、闭包 ```javascript var x = 10; // 全局作用域A中的变量x function fn1() { // 局部作用域fn1 var x = 20; function fn2() { // 局部作用域fn2 console.log(x); } return fn2; } var fn3 = fn1(); fn3(); // 20 ``` 由（二）我们可以知道结果为20是对的，其实我们将这种在一个局部作用域中使用另外一个局部作用域中的私有变量的行为成为闭包（也就是在一个函数中使用另外一个函数中的通过var声明的私有变量的行为称为闭包）。闭包就是将函数内部和函数外部连接起来的一座桥梁。 四、哪些方式可以形成闭包呢？ 很明显，由（三）可以知道，通过在函数中return一个函数的方式可以形成闭包，这种方式是将函数当做返回值。还有别的方式吗？ 另外一种方式就是返回一个对象，对象中函数方法（即函数）： ```javascript function returnObj() { var num = 800; return { objFn: function() { console.log(num++); } } } var returnObj1 = returnObj(); returnObj1.objFn();// 800 returnObj1.objFn();// 801 var returnObj2 = returnObj(); returnObj2.objFn();// 800 returnObj2.objFn();// 801 returnObj2.objFn();// 802 // 或者 new + 构造函数的隐性return返回 function Persion() { var num = 600; this.say = function() { console.log(num++); } } var p1 = new Persion(); p1.say(); //600 p1.say();//601 p1.say();//602 var p2 = new Persion(); p2.say(); //600 p2.say(); //601 p2.say(); //602 ``` 形成闭包的两种方式：函数作为返回值、将含有方法的对象作为返回值。 五、闭包的用途 闭包作用： 1、将局部作用域中的局部变量像全局作用域中的全局变量一样保存在内存中不被销毁； 2、闭包的每一次形成局部作用域中的局部变量都被重新声明定义了一次，与之前声明定义的局部变量是不一样的，是独立的（即使变量名一样），就好像new+构造函数一样造了很多示例。 ```javascript var x = 10; // 全局作用域A中的变量x function fn1() { // 局部作用域fn1 var x = 20; xAdd = function() { x++; }; function fn2() { // 局部作用域fn2 console.log(x); } return fn2; } var fn3 = fn1(); fn3(); // 20 xAdd(); fn3(); // 21 xAdd(); fn3(); // 22 ``` 由结果可以知道函数f1中的局部变量x一直保存在内存中，并没有在f1调用后被自动清除。 为什么会这样呢？原因就在于f1是f2的父函数，而f2被赋给了一个全局变量，这导致f2始终在内存中，而f2的存在依赖于f1，因此f1也始终在内存中，不会在调用结束后，被垃圾回收机制（garbage collection）回收。 这段代码中另一个值得注意的地方，就是"xAdd=function(){x++;}"这一行，首先在xAdd前面没有使用var关键字，因此xAdd是一个全局变量，而不是局部变量。其次，xAdd的值是一个匿名函数（anonymous function），而这个匿名函数本身也是一个闭包，所以xAdd相当于是一个setter，可以在函数外部对函数内部的局部变量进行操作。 六、使用闭包的注意点 1）由于闭包会使得函数中的变量都被保存在内存中，内存消耗很大，所以不能滥用闭包，否则会造成网页的性能问题，在IE中可能导致内存泄露。解决方法是，在退出函数之前，将不使用的局部变量全部删除。 2）闭包会在父函数外部，改变父函数内部变量的值。 ```javascript var x = 10; // 全局作用域A中的变量x function fn1() { // 局部作用域fn1 var x = 20; function fn2() { // 局部作用域fn2 console.log(x); x++; } return fn2; } var fn3 = fn1(); fn3(); // 20 fn3(); // 21 fn3(); // 22 ``` 如果你把父函数当作对象（object）使用，把闭包当作它的公用方法（Public Method），把内部变量当作它的私有属性（private value），这时一定要小心，不要随便改变父函数内部变量的值。 ```javascript var name = "The Window"; 一、自由变量

<scriptSS>

var x = 10; // 全局作用域A中的变量x
function fn1() {  // 局部作用域fn1

    var y = 20;
    console.log(x);

   function fn2() { // 局部作用域fn2
        console.log(y);
   }
 }

</script>

作用域链：全局作用域A ==> 局部作用域fn1 ==> 局部作用域fn2

我们称在局部作用域中通过作用域链使用的变量为自由变量，x是fn1中的自由变量，y是fn2中的自由变量。

二、局部作用域中自由变量的指向

<scriptSS>

   var x = 10; // 全局作用域A中的变量x
    function fn1() {  // 局部作用域fn1
        console.log(x);
    }

    var fn3 = function() { // 局部作用域fn3
        var x = 30;

        fn1();

    }

    fn3(); // 10

</script>

在局部作用域形成(或者说定义)时，局部作用域中自由变量的指向已经确定，与在哪里调用执行无关。

三、闭包

<scriptSS>

    var x = 10; // 全局作用域A中的变量x

    function fn1() {  // 局部作用域fn1
        var x = 20;

        function fn2() { // 局部作用域fn2
            console.log(x);
        }

        return fn2;

    }

    var fn3 = fn1();

    fn3(); // 20


</script>

由（二）我们可以知道结果为20是对的，其实我们将这种在一个局部作用域中使用另外一个局部作用域中的私有变量的行为成为闭包（也就是在一个函数中使用另外一个函数中的通过var声明的私有变量的行为称为闭包）。闭包就是将函数内部和函数外部连接起来的一座桥梁。

四、哪些方式可以形成闭包呢？

很明显，由（三）可以知道，通过在函数中return一个函数的方式可以形成闭包，这种方式是将函数当做返回值。还有别的方式吗？

另外一种方式就是返回一个对象，对象中函数方法（即函数）：

<scriptSS>


function returnObj() {
  var num = 800;
  return {
      objFn: function() {
        console.log(num++);
      }
  }
}

var returnObj1 = returnObj();
returnObj1.objFn();// 800
returnObj1.objFn();// 801

var returnObj2 = returnObj();
returnObj2.objFn();// 800
returnObj2.objFn();// 801
returnObj2.objFn();// 802

// 或者 new + 构造函数的隐性return返回

function Persion() {
    var num = 600;
    this.say = function() {
        console.log(num++);
    }
}
var p1 = new Persion();
p1.say(); //600 
p1.say();//601 
p1.say();//602

var p2 = new Persion();
p2.say(); //600 
p2.say(); //601 
p2.say(); //602 

</script>

形成闭包的两种方式：函数作为返回值、将含有方法的对象作为返回值。

五、闭包的用途

闭包作用： 1、将局部作用域中的局部变量像全局作用域中的全局变量一样保存在内存中不被销毁； 2、闭包的每一次形成局部作用域中的局部变量都被重新声明定义了一次，与之前声明定义的局部变量是不一样的，是独立的（即使变量名一样），就好像new+构造函数一样造了很多示例。

<scriptSS>

   var x = 10; // 全局作用域A中的变量x

    function fn1() {  // 局部作用域fn1
        var x = 20;

        xAdd = function() { x++; };

        function fn2() { // 局部作用域fn2
            console.log(x);
        }

        return fn2;

    }

    var fn3 = fn1();

    fn3(); // 20

    xAdd();

    fn3(); // 21

    xAdd();

    fn3(); // 22


</script>

由结果可以知道函数f1中的局部变量x一直保存在内存中，并没有在f1调用后被自动清除。

为什么会这样呢？原因就在于f1是f2的父函数，而f2被赋给了一个全局变量，这导致f2始终在内存中，而f2的存在依赖于f1，因此f1也始终在内存中，不会在调用结束后，被垃圾回收机制（garbage collection）回收。

这段代码中另一个值得注意的地方，就是"xAdd=function(){x++;}"这一行，首先在xAdd前面没有使用var关键字，因此xAdd是一个全局变量，而不是局部变量。其次，xAdd的值是一个匿名函数（anonymous function），而这个匿名函数本身也是一个闭包，所以xAdd相当于是一个setter，可以在函数外部对函数内部的局部变量进行操作。

六、使用闭包的注意点

1）由于闭包会使得函数中的变量都被保存在内存中，内存消耗很大，所以不能滥用闭包，否则会造成网页的性能问题，在IE中可能导致内存泄露。解决方法是，在退出函数之前，将不使用的局部变量全部删除。 2）闭包会在父函数外部，改变父函数内部变量的值。

<scriptSS>

    var x = 10; // 全局作用域A中的变量x

    function fn1() {  // 局部作用域fn1
        var x = 20;

        function fn2() { // 局部作用域fn2
            console.log(x);
            x++;
        }

        return fn2;

    }

    var fn3 = fn1();

    fn3(); // 20

    fn3(); // 21

    fn3(); // 22


</script>

如果你把父函数当作对象（object）使用，把闭包当作它的公用方法（Public Method），把内部变量当作它的私有属性（private value），这时一定要小心，不要随便改变父函数内部变量的值。

var name = "The Window";

　　　　return function(){

　　　　};

　　}
};
console.log(object.getNameFunc()());

结果：The Window。ruturn的匿名函数中的this执行window。

　　name : "My Object",

        var that = this;

　　　　　　return that.name;
　　　　};

　　}

七、总结

1、自由变量

闭包的形成是利用在局部作用域形成(或者说定义)时，局部作用域中自由变量的指向已经确定，与在哪里调用执行无关这一特性形成的（明白这一点即使你不懂闭包也不会出错，甚至可以说你完全不用去理解关于闭包的一切）。
